2)Data la descrizione di un ruolo (compilatore) e dato l'ID_ENTE (10) trovare nome e cognome di tutti i funzionari che hanno il 
  ruolo di compilatore per l'ente indicato
  
	--  select TD.DESCRIZIONE,TE.ID_ENTE,TF.NOME, TF.COGNOME
		from GAGREA_SHARED.TENTE_SOP TE, GAGREA.TFUNZIONARIO TF, GAGREA.TORGANICO TOR, GAGREA.TDECODIFICA_SOP TD
		where TE.ID_ENTE=10 and TD.DESCRIZIONE='COMPILATORE' and
		TF.ID_FUNZIONARIO = TOR.ID_FUNZIONARIO and 
		TOR.ID_ENTE = TE.ID_ENTE and 
		TD.SOTTO_CODICE = TOR.SCO_RUOLO
